### 冒泡排序

```
BubbleSort(A[], n) {
	for i = 0 to n - 2
		for j = 0 to n - 2 - i
			if A[j] > A[j + 1]
				swap(A[j], A[j + 1])
}
```

### 选择排序

```
SelectionSort(A[], n) {
	for i = 0 to n - 1
		index = i
		for j = i + 1 to n - 1
			if A[j] < A[index]
				index = j
		if index != i
			swap(A[index], A[i])
}
```

### 插入排序

```
InsertionSort(A[], n) {
	for i = 1 to n - 1
		key = A[i]
		j = i - 1
		while j > 0 and A[j] < A[j - 1]
			A[j] = A[j - 1]
			j = j - 1
		A[j] = key
}
```

### 地精排序

```
GnomeSort(A[], n) {
	i = 0
	while i < n
		if i == 0 || A[i] >= A[i - 1]
			i = i + 1
		else
			swap(A[i], A[i - 1])
			i = i - 1
}
```

### 希尔排序

```
ShellSort(A[], n) {
	gap = n / 2
	while gap > 0
		for i = gap to n - 1
			key = A[i]
			j = i - gap
			while j > gap and A[j] < A[j - gap]
				A[j] = A[j - gap]
				j = j - gap
			A[j] = key
		gap = gap / 2
}
```

ParitySort(A[], n) {
	while not A.sorted
		for i = 0 to n - 2 by 2
			if A[i] > A[i + 1]
				swap(A[i], A[i + 1])
		for i = 1 to n - 2 by 2

Median3(A, low, high) {
	center = (low + high) / 2
	SortBy(A[low], A[center], A[high])
	swap(A[center], A[high - 1])
	return A[high - 1]
}

QSort(A[], low, high) {
	if low >= high
		return
	pivot = Median3(A, low, high)
	i = low
	j = high - 1
	while true
		repeat
			i = i + 1
		until A[i] >= pivot
	 	repeat
	 		j = j - 1

	 	else

	 swap(A[i], A[high - 1])	/* A[i] expected to be an element >= pivot */

### 归并排序

```
MergeSort(A[], n) {
	MSort(A, 0, n - 1)
}

MSort(A[], low, high) {
	if low == high
		return
	mid = (low + high) / 2
	MSort(A, low, mid)
	MSort(A, mid + 1, high)
	Merge(A, low, mid, high)
}

Merge(A, low, mid, high) {
	i = low, j = mid + 1, k = 0
	while i <= mid and j <= high
		if A[i] <= A[j]
			C[k] = A[i]
			k = k + 1
			i = i + 1
		else
			C[k] = A[j]
			k = k + 1
			j = j + 1
	while i <= mid
		C[k] = A[i]
		k = k + 1
		i = i + 1
	while j <= high
		C[k] = A[j]
		k = k + 1
		j = j + 1
	A[low..high] = C[0..k-1]
}
```

BucketSort(A[], n) {
	Let B[0..n - 1] be a new list array
	factor = n / MaxElement(A)
	for i = 0 to n - 1
		Insert(A[i], B[A[i] / factor])

	A = ConcatenateLists(B)

### 	计数排序

```
CountingSort(A[], n, Range) {
	Let C[Range] be a new array
	for e in A
		C[e] = C[e] + 1
	j = 0
	for i = 0 to Range
		while C[i] > 0
			C[i] = C[i] - 1
			A[j] = i
			j = j + 1
}
```

### 双调排序

```
BitonicSort(A[], low, high, reversed) {
	if low < high
		mid = (low + high) / 2
		BitonicSort(A, low, mid, reversed)
		BitonicSort(A, mid + 1, high, not reversed)
		BitonicMerge(A, low, high, reversed)
}

BitonicMerge(A[], low, high, reversed) {
	mid = (low + high) / 2
	if low < high
		for i = low to mid
			if not reversed
				if A[i] > A[i + mid]
					swap(A[i], A[i + mid])
			else
				if A[i] < A[i + mid]
					swap(A[i], A[i + mid])
	BitonicMerge(A, low, mid, reversed)
	BitonicMerge(A, mid + 1, high, reversed)
}
```
